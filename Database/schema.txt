--schema used to create database

CREATE KEYSPACE IF NOT EXITSTS hash WITH REPLICATION = {'class': 'SimpleStrategy', 'replication_factor':1};

use hash;

CREATE TABLE public_recipe (
	id text,
	name text,
	time int,
	imageurl text,
	ingredients list<text>,
	numberofserving int,
	flavor text,
	instruction text,
	PRIMARY KEY(id) );

CREATE TABLE private_recipe (
	id uuid,
	name text,
	time int,
	imageurl text,
	ingredients list<text>,
	numberofserving int,
	flavor text,
	instruction text,
	creator uuid,
	PRIMARY KEY(id) );
	
CREATE TABLE user (
	id uuid,
	username text,
	email text,
	password text,
	favourite list<text>,
	own_recipe list<uuid>,
	PRIMARY KEY(id) );
